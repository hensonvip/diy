SELECT g.goods_id, g.goods_name, g.supplier_id, g.is_shipping, g.click_count, g.is_promote, g.goods_number, g.goods_total, g.market_price,g.cat_id,g.brand_id, g.is_new, g.is_best, g.is_hot, g.shop_price AS org_price, g.user_id,  IFNULL(mp.user_price, g.shop_price * '1') AS shop_price, g.promote_price,  IF(g.promote_price != ''  AND g.promote_start_date < 1530771399 AND g.promote_end_date > 1530771399, g.promote_price, shop_price)  AS shop_p, g.goods_type,  g.promote_start_date, g.promote_end_date, g.goods_brief, g.goods_thumb, g.goods_img, g.best_img  FROM `xueyou`.`hunuo_goods` AS g  LEFT JOIN `xueyou`.`hunuo_member_price` AS mp  ON mp.goods_id = g.goods_id  AND mp.user_rank = '2'  WHERE g.is_on_sale = 1 AND g.is_alone_sale = 1 AND g.is_delete = 0 AND goods_status IN (0,4)   AND (g.cat_id  IN ('0','85','86','87','88')  OR g.goods_id IN ('') )   and (g.is_real = 0 or g.is_real =1)   ORDER BY g.sort_order desc
